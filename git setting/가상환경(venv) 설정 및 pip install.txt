# venv 가상환경 컨트롤 하는 방법

# 시스템 단위로 Homebrew로 Python 설치 (글로벌)
brew install python@3.13

# 프로젝트 폴더로 이동해서 가상환경 생성
/opt/homebrew/bin/python3 -m venv encar_gon

# 가상환경 활성화
source encar_gon/bin/activate

# 이제 venv 전용으로 pip 설치
pip install pandas numpy matplotlib

# 가상환경 내에 설치된 pip 리스트를 requirements 문서로 내보내기
pip freeze > requirements.txt

# 가상환경 비활성화
deactivate

# 다른 pc에서 레포 따서 작업할때 가상환경 세팅 및 pip install 을 requirements.txt 에 있는걸로 진행하고 싶을때 편하게
python3 -m venv <폴더이름>
source .<폴더이름>/bin/activate
pip install -r requirements.txt

(혹은 필요한게 있으면 이렇게 또 추가로) pip install pandas numpy matplotlib
deactivate
